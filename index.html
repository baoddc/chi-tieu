    Liên kết dữ liệu Google Sheet

    Kết nối ứng dụng với Google Sheets của bạn để tự động lưu trữ và đồng bộ hóa mọi giao dịch trực tuyến.

    Quản lý Ví tài khoản

    Danh sách các ví/tài khoản thanh toán của bạn. Các thay đổi về tên ví sẽ tự động cập nhật các giao dịch liên quan.

    Hướng dẫn thiết lập cơ sở dữ liệu (5 phút)

    BƯỚC 1: Tạo Google Sheet
    1. Truy cập sheets.new để tạo một Google Sheet trống.
    2. Đặt tên cho Google Sheet (Ví dụ: Sổ Quản Lý Chi Tiêu Pro).
    BƯỚC 2: Cài đặt mã nguồn Google Apps Script
    1. Trong Google Sheet mới tạo, nhấn vào Extensions (Tiện ích mở rộng) > Apps Script.
    2. Xóa mọi đoạn mã có sẵn trong khung editor.
    3. Bấm nút Sao chép mã nguồn bên dưới, dán toàn bộ vào Apps Script và bấm Save (Lưu).

var TABLES = {
  "Transactions": ["id", "date", "amount", "category", "type", "note", "wallet"],
  "Categories": ["id", "name", "type", "color", "icon", "budget"],
  "Wallets": ["id", "name", "color", "icon"],
  "Debts": ["id", "lender", "type", "amount", "date", "dueDate", "note", "wallet"]
};

function doGet(e) {
  try {
    var data = readAllData();
    return createJsonResponse({ success: true, data: data });
  } catch (error) {
    return createJsonResponse({ success: false, error: error.toString() });
  }
}

function doPost(e) {
  try {
    if (!e.postData || !e.postData.contents) {
      throw new Error("No payload received");
    }
    var request = JSON.parse(e.postData.contents);
    var action = request.action;
    var tableName = request.table;
    var payload = request.data;
    if (!TABLES[tableName]) throw new Error("Bảng không hợp lệ: " + tableName);
    
    var result;
    if (action === "readAll") {
      result = readAllData();
    } else if (action === "insertRow") {
      result = insertRow(tableName, payload);
    } else if (action === "updateRow") {
      result = updateRow(tableName, payload.id, payload);
    } else if (action === "deleteRow") {
      result = deleteRow(tableName, payload.id);
    } else if (action === "syncAll") {
      result = syncAllData(payload);
    } else {
      throw new Error("Hành động không được hỗ trợ");
    }
    return createJsonResponse({ success: true, data: result });
  } catch (error) {
    return createJsonResponse({ success: false, error: error.toString() });
  }
}

function createJsonResponse(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateSheet(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    var headers = TABLES[name];
    sheet.appendRow(headers);
    var range = sheet.getRange(1, 1, 1, headers.length);
    range.setFontWeight("bold").setBackground("#f3f4f6").setFontColor("#374151");
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function readAllData() {
  var result = {};
  for (var tableName in TABLES) {
    var sheet = getOrCreateSheet(tableName);
    var headers = TABLES[tableName];
    var data = [];
    var lastRow = sheet.getLastRow();
    if (lastRow > 1) {
      var values = sheet.getRange(2, 1, lastRow - 1, headers.length).getValues();
      for (var r = 0; r < values.length; r++) {
        var rowObj = {};
        for (var c = 0; c < headers.length; c++) {
          rowObj[headers[c]] = values[r][c];
        }
        data.push(rowObj);
      }
    }
    result[tableName] = data;
  }
  return result;
}

function insertRow(tableName, rowData) {
  var sheet = getOrCreateSheet(tableName);
  var headers = TABLES[tableName];
  var rowValues = [];
  for (var i = 0; i < headers.length; i++) {
    var val = rowData[headers[i]];
    rowValues.push(val !== undefined ? val : "");
  }
  sheet.appendRow(rowValues);
  return rowData;
}

function updateRow(tableName, id, rowData) {
  if (!id) throw new Error("Yêu cầu ID.");
  var sheet = getOrCreateSheet(tableName);
  var headers = TABLES[tableName];
  var lastRow = sheet.getLastRow();
  if (lastRow <= 1) return insertRow(tableName, rowData);
  var idValues = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
  var rowIndex = -1;
  for (var i = 0; i < idValues.length; i++) {
    if (String(idValues[i][0]) === String(id)) {
      rowIndex = i + 2;
      break;
    }
  }
  if (rowIndex === -1) return insertRow(tableName, rowData);
  for (var c = 0; c < headers.length; c++) {
    var key = headers[c];
    if (rowData[key] !== undefined) {
      sheet.getRange(rowIndex, c + 1).setValue(rowData[key]);
    }
  }
  return rowData;
}

function deleteRow(tableName, id) {
  if (!id) throw new Error("Yêu cầu ID.");
  var sheet = getOrCreateSheet(tableName);
  var lastRow = sheet.getLastRow();
  if (lastRow <= 1) return false;
  var idValues = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
  var rowIndex = -1;
  for (var i = 0; i < idValues.length; i++) {
    if (String(idValues[i][0]) === String(id)) {
      rowIndex = i + 2;
      break;
    }
  }
  if (rowIndex !== -1) {
    sheet.deleteRow(rowIndex);
    return true;
  }
  return false;
}

function syncAllData(payload) {
  for (var tableName in payload) {
    if (!TABLES[tableName]) continue;
    var sheet = getOrCreateSheet(tableName);
    var lastRow = sheet.getLastRow();
    if (lastRow > 1) {
      sheet.deleteRows(2, lastRow - 1);
    }
    var items = payload[tableName] || [];
    var headers = TABLES[tableName];
    if (items.length > 0) {
      var rowsToAppend = [];
      for (var i = 0; i < items.length; i++) {
        var item = items[i];
        var row = [];
        for (var c = 0; c < headers.length; c++) {
          var val = item[headers[c]];
          row.push(val !== undefined ? val : "");
        }
        rowsToAppend.push(row);
      }
      sheet.getRange(2, 1, rowsToAppend.length, headers.length).setValues(rowsToAppend);
    }
  }
  return readAllData();
}
    BƯỚC 3: Triển khai thành Web App
    1. Trong Apps Script, bấm vào nút Deploy (Triển khai) góc trên bên phải > chọn New deployment (Triển khai mới).
    2. Chọn loại triển khai là Web App (Ứng dụng web).
    3. Mục Execute as: Chọn Me (Tài khoản Google của bạn).
    4. Mục Who has access: Chọn Anyone (Bất kỳ ai).
    5. Bấm Deploy. Nhấp vào Authorize access và cấp quyền cho tài khoản Google của bạn nếu được yêu cầu.
    6. Sao chép đường dẫn Web app URL, dán vào ô nhập liệu bên trên và bấm Kiểm tra & Kết nối.
